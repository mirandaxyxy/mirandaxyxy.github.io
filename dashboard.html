Filter all days with snow and precipitation in 2009 and 2010, and with non null values for temperatures.

data("ny_noaa")

noaa = ny_noaa %>% 
  drop_na(tmax,tmin,snow,prcp) %>% 
  mutate(tmax = as.numeric(tmax)/10, tmin = as.numeric(tmin)/10, prcp = as.numeric(prcp)/10) %>% 
  filter(snow > 0 & prcp > 0 & year(date) >= 2009) %>% 
  select(-snwd)
